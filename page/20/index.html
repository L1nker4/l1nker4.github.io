mkdir notes

# 初始化一个git仓库 
git init

echo "my notes" > readme.md

# 查看状态
git status

# 查看修改内容
git diff readme.md

# 添加文件到仓库
git add readme.md

# 提交更改
git commit -m "add readme"

# 查看提交日志
git log

# 回退到上一个版本
git reset --hard HEAD^

# HEAD: 当前版本
# HEAD^: 上一个版本
# HEAD~100: 前一百个版本
# 58a65b: 指定版本

# 在本地创建一个 gitlearn 目录，并进入
mkdir gitlearn
cd gitlearn

# 使用命令 git init 初始化一个库
git init

# 添加远程库 origin
git remote add origin git@github.com:l1nker4/gitlearn.git

# 添加 readme.md 文件
echo "git learn is very easy!" > readme.md

# 添加文件并提交到仓库
git add readme.md
git commit -m "add readme"

# 执行向远程仓库 origin master 分支推送更新（-u 参数用于关联远程仓库）
git push -u origin master

# 切换分支

git checkout -b dev

git merge dev 

# 删除分支
git branch -d dev
