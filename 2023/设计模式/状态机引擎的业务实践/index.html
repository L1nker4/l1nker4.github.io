зҠ¶жҖҒжңәеј•ж“Һзҡ„дёҡеҠЎе®һи·ө

еүҚиЁҖ

еңЁејҖеҸ‘дёӯйҖҡеёёдјҡжңүзҠ¶жҖҒеұһжҖ§пјҢдҫӢеҰӮи®ўеҚ•зҠ¶жҖҒгҖҒдәӢд»¶еӨ„зҪ®зҠ¶жҖҒгҖҒOAиҜ·еҒҮеҚ•зҠ¶жҖҒзӯүпјҢйғҪдјҡж №жҚ®дёҚеҗҢзҡ„еҠЁдҪңпјҢеҺ»жӣҙж–°еҜ№еә”зҡ„зҠ¶жҖҒгҖӮеҰӮдёӢпјҡ

int status = 0;  
if (condition1){  
    status = 1;  
}else if (condition2){  
    status = 2;  
}else if (condition3){  
    status = 3;  
}

дёҠиҝ°зӨәдҫӢеӯҳеңЁзҡ„й—®йўҳпјҡ - еңЁйңҖжұӮеўһеҠ зҡ„жғ…еҶөдёӢпјҢif.elseдјҡдёҚж–ӯиҶЁиғҖ - д»Јз ҒеҸҜиҜ»жҖ§е·®пјҢз•Ҙеҫ®ж”№еҠЁдјҡеҜјиҮҙеҗ„з§Қй—®йўҳ - е…¶д»–дёҡеҠЎйҖ»иҫ‘дјҡиҖҰеҗҲеңЁif.elseд»Јз Ғж®өдёӯ

й’ҲеҜ№зі»з»ҹдёӯзҠ¶жҖҒзҡ„з®ЎзҗҶпјҢеҸҜд»ҘдҪҝз”ЁжңүйҷҗзҠ¶жҖҒжңәеҺ»и§ЈеҶіпјҢжңүйҷҗзҠ¶жҖҒжңәжҳҜиЎЁзӨәжңүйҷҗдёӘзҠ¶жҖҒд»ҘеҸҠзҠ¶жҖҒй—ҙиҪ¬з§»зҡ„жЁЎеһӢгҖӮзҠ¶жҖҒжңәз”ұдәӢд»¶гҖҒзҠ¶жҖҒгҖҒеҠЁдҪңдёүйғЁеҲҶз»„жҲҗгҖӮ

зҠ¶жҖҒжңәзҡ„з®ҖеҚ•е®һзҺ°

жқЎд»¶еҲҶж”ҜеҲӨж–ӯ

жңҖз®ҖеҚ•зҡ„ж–№жЎҲжҳҜйҖҡиҝҮжқЎд»¶еҲҶж”ҜжҺ§еҲ¶зҠ¶жҖҒзҡ„иҪ¬з§»пјҢиҝҷйҖӮеҗҲдәҺдёҡеҠЎзҠ¶жҖҒиҫғе°‘пјҢе№¶дё”зҠ¶жҖҒи·іиҪ¬зҡ„йҖ»иҫ‘иҫғз®ҖеҚ•зҡ„еңәжҷҜпјҢдҪҶжҳҜеҪ“и§ҰеҸ‘дәӢд»¶иҫғеӨҡж—¶пјҢйңҖиҰҒеөҢеҘ—еӨҡеұӮжқЎд»¶еҲӨж–ӯпјҢеҪ“йңҖжұӮдёӯзҠ¶жҖҒеҸҳжӣҙдәӢд»¶ж”№еҸҳж—¶пјҢйңҖиҰҒж”№еҠЁзҡ„йҖ»иҫ‘иҫғеӨ§гҖӮжЎҲдҫӢд»Јз ҒеҰӮдёӢпјҡ

public enum ActivityState {

    PREPARE(0),
    STARTED(1),
    END(2)
    ;

    public final int status;


    ActivityState(int status) {
        this.status = status;
    }

    public int getStatus() {
        return status;
    }
}

/**
 * жҙ»еҠЁзҠ¶жҖҒ-зҠ¶жҖҒжңә
 * @author l1nker4
 */
public class ActivityStateMachine {
    /**
     * жҙ»еҠЁзҠ¶жҖҒ
     */
    private ActivityState currentState;

    public ActivityStateMachine() {
        this.currentState = ActivityState.PREPARE;
    }


    /**
     * жҙ»еҠЁејҖе§Ӣ
     */
    public void begin() {
        if (currentState.equals(ActivityState.PREPARE)) {
            this.currentState = ActivityState.STARTED;
            //do something....
        }
    }

    /**
     * жҙ»еҠЁз»“жқҹ
     */
    public void end(){
        if (currentState.equals(ActivityState.STARTED)){
            this.currentState = ActivityState.END;
            //do something...
        }
    }

}

DSLж–№жЎҲ

DSLпјҲDomain-Specific Languagesпјүпјҡй’ҲеҜ№д»Ҙзү№е®ҡйўҶеҹҹзҡ„иҜӯиЁҖпјҢз”ЁдәҺи§ЈеҶідёҖзұ»й—®йўҳпјҢдҫӢеҰӮпјҡжӯЈеҲҷиЎЁиҫҫејҸеҸҜд»Ҙи§ЈеҶіеӯ—з¬ҰдёІеҢ№й…ҚиҝҷдёҖзұ»й—®йўҳгҖӮеҸҜеҲҶдёәд»ҘдёӢдёӨз§Қзұ»еһӢпјҡ

  • Internal DSLпјҡдёҡеҠЎд»Јз ҒйҖҡиҝҮд»Јз ҒзӣҙжҺҘиҝӣиЎҢй…ҚзҪ®пјҢJava Internal DSLдёҖиҲ¬зӣҙжҺҘеҲ©з”Ёе»әйҖ иҖ…жЁЎејҸе’ҢжөҒејҸжҺҘеҸЈжһ„е»ә
  • External DSLпјҡйҖҡиҝҮеӨ–йғЁй…ҚзҪ®ж–Үд»¶пјҢиҝҗиЎҢж—¶иҜ»еҸ–иҝӣиЎҢи§ЈжһҗпјҢдҫӢеҰӮXMLгҖҒJSONзӯүж јејҸ

COLA

COLAжҳҜйҳҝйҮҢе·ҙе·ҙејҖжәҗзҡ„DSLзҠ¶жҖҒжңәе®һзҺ°пјҢж №жҚ®дҪңиҖ…зҡ„д»Ӣз»ҚпјҢзӣёжҜ”иҫғе…¶д»–ејҖжәҗзҡ„зҠ¶жҖҒжңәе®һзҺ°пјҲSpring StatemachineзӯүпјүпјҢе…¶дјҳзӮ№еҰӮдёӢпјҡ

  • ејҖжәҗзҡ„зҠ¶жҖҒжңәеј•ж“ҺжҖ§иғҪиҫғе·®пјҢзҠ¶жҖҒжңәжңүзҠ¶жҖҒпјҢйңҖиҰҒиҖғиҷ‘зәҝзЁӢе®үе…Ё
  • ејҖжәҗзҡ„зҠ¶жҖҒжңәиҫғдёәеӨҚжқӮпјҢеӨӘйҮҚгҖӮ

йўҶеҹҹжЁЎеһӢ

COLAзҠ¶жҖҒжңәзҡ„ж ёеҝғжҰӮеҝөжңүпјҡ

  • StateпјҡзҠ¶жҖҒ
  • EventпјҡзҠ¶жҖҒз”ұдәӢд»¶и§ҰеҸ‘
  • TransitionпјҡзҠ¶жҖҒй—ҙиҪ¬жҚўзҡ„иҝҮзЁӢ
  • External TransitionпјҡstateA -> stateB
  • Internal TransitionпјҡsteteA -> stateA
  • ConditionпјҡжҳҜеҗҰе…Ғи®ёеҲ°иҫҫжҹҗз§ҚзҠ¶жҖҒ
  • ActionпјҡstateеҸҳеҢ–ж—¶жү§иЎҢзҡ„еҠЁдҪң

public enum Events {

    STARTED("activity started"),

    END("activity ended")
    ;


    private String event;

    Events(String event) {
        this.event = event;
    }

    public void setEvent(String event) {
        this.event = event;
    }
}

@Test
public void testMachine() {
        StateMachineBuilder<ActivityState, Events, Object> builder = StateMachineBuilderFactory.create();

        builder.externalTransition()
                .from(ActivityState.PREPARE)
                .to(ActivityState.STARTED)
                .on(Events.STARTED)
                .when(checkCondition())
                .perform(doAction());
        StateMachine<ActivityState, Events, Object> machine = builder.build("test");
        ActivityState activityState = machine.fireEvent(ActivityState.PREPARE, Events.STARTED, new Object());
        log.info("new state: {}", activityState);
    }


    private Condition<Object> checkCondition() {
        return (context -> {return true;});
    }

    private Action<ActivityState, Events, Object> doAction() {
        return ((from, to, event, context) -> {
            log.info("{} is doing, from {}, to {}, on: {}", context, from, to, event);
        });
    }
